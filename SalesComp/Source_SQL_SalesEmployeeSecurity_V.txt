SELECT     HR.REPORTING_ACCESS_LIST_L.PRIMARY_EMAIL_ADDRESS_TX, 
                      TO_NUMBER(OLAP_ETL.SALES_PERSON_NORMALIZED_V.SALES_PERSON_EMPLOYEE_KY) AS SALES_EMPLOYEE_KY, 
                      HR.REPORTING_ACCESS_LIST_L.SECURITY_ACCESS_CD
FROM         OLAP_ETL.SALES_PERSON_NORMALIZED_V INNER JOIN
                      HR.REPORTING_ACCESS_LIST_L ON 
                      OLAP_ETL.SALES_PERSON_NORMALIZED_V.SALES_MGR_EMPLOYEE_KY = HR.REPORTING_ACCESS_LIST_L.CONTEXT_LEVEL_VALUE_KY INNER JOIN
                      HR.EMPLOYEE_D ON HR.REPORTING_ACCESS_LIST_L.PRIMARY_EMAIL_ADDRESS_TX = HR.EMPLOYEE_D.PRIMARY_EMAIL_ADDRESS_TX
WHERE     (HR.REPORTING_ACCESS_LIST_L.SECURITY_CONTEXT_NM = 'SALES_COMPENSATION') AND (HR.EMPLOYEE_D.HP_STATUS_NM = 'Active') AND 
                      (HR.REPORTING_ACCESS_LIST_L.CONTEXT_LEVEL_NM = 'SALES_EMPLOYEE_FULL_NM')
