SELECT     HR.REPORTING_ACCESS_LIST_L.PRIMARY_EMAIL_ADDRESS_TX, TO_NUMBER(OLAP_ETL.SALES_GEOGRAPHY_NORMALIZED_V.GEOGRAPHY_KY) 
                      AS SALES_GEOGRAPHY_KY, HR.REPORTING_ACCESS_LIST_L.SECURITY_ACCESS_CD
FROM         OLAP_ETL.SALES_GEOGRAPHY_NORMALIZED_V INNER JOIN
                      HR.REPORTING_ACCESS_LIST_L ON 
                      OLAP_ETL.SALES_GEOGRAPHY_NORMALIZED_V.PARENT_GEOGRAPHY_CD = HR.REPORTING_ACCESS_LIST_L.CONTEXT_LEVEL_VALUE_NM INNER
                       JOIN
                      HR.EMPLOYEE_D ON HR.REPORTING_ACCESS_LIST_L.PRIMARY_EMAIL_ADDRESS_TX = HR.EMPLOYEE_D.PRIMARY_EMAIL_ADDRESS_TX
WHERE     (HR.REPORTING_ACCESS_LIST_L.SECURITY_CONTEXT_NM = 'SALES_COMPENSATION') AND 
                      (HR.REPORTING_ACCESS_LIST_L.CONTEXT_LEVEL_NM = 'SALES_GEOGRAPHY_NM') AND (HR.EMPLOYEE_D.HP_STATUS_NM = 'Active')
